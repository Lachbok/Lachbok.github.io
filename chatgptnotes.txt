1. fetch() – Load text or JSON files
fetch('posts/post1.txt')
  .then(response => response.text())
  .then(data => {
    document.getElementById('blog').innerHTML += `<div class="post">${data}</div>`;
  });

2. innerHTML – Insert content into the page
document.getElementById('blog').innerHTML = "<p>Hello</p>";

3. forEach Loop – Load multiple posts
const posts = ['post1.txt', 'post2.txt'];
posts.forEach(filename => {
  fetch(`posts/${filename}`)
    .then(res => res.text())
    .then(data => {
      document.getElementById('blog').innerHTML += `<div class="post">${data}</div>`;
    });
});

4. Optional: JSON post format
// posts/post1.json
{
  "title": "My First Post",
  "date": "2025-06-08",
  "content": "This is the body of the post!"
}
fetch('posts/post1.json')
  .then(res => res.json())
  .then(post => {
    document.getElementById('blog').innerHTML += `
      <div class="post">
        <h2>${post.title}</h2>
        <p><em>${post.date}</em></p>
        <p>${post.content}</p>
      </div>`;
  });
